~version: "2.1"

# Identificar as páginas de artigos estáticos em qualquer subdiretório
?path: .+/posts/.+\.html

# Definir o corpo do artigo
body: //article[@id="article-content"]

# Definir o título
title: //h1[1]
@remove: $title

# Telegram IV não suporta blocos colapsáveis interativos (<details>)
# Vamos extrair o conteúdo do summary e do details para o corpo principal
@rename(h3): $body//details/summary
@inline: $body//details

# Ajustar imagens
image_url: $body//img/@src

# Remover elementos desnecessários
@remove: //nav
@remove: //footer
@remove: //script

# Metadados básicos para IV
author: "Moonshot Blog"

# Limpeza final
@debug: $body

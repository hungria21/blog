~version: "2.1"

# Identificar as páginas de artigos estáticos em qualquer subdiretório (suporte a GitHub Pages com repo name)
?path: .+/posts/.+\.html

# Definir o corpo do artigo
body: //article[@id="article-content"]

# Definir o título
title: //h1[1]
@remove: $title

# Telegram IV não suporta blocos colapsáveis interativos (<details>)
# Vamos garantir que o conteúdo seja visível transformando o resumo em um cabeçalho
@wrap(<h3 inner>): $body//details/summary
@combine(<p>): $body//details/p

# Ajustar imagens
image_url: $body//img/@src

# Remover elementos desnecessários
@remove: //nav
@remove: //footer
@remove: //script

# Metadados básicos para IV
author: "Moonshot Blog"
published_date: //span[@class="date"]

# Limpeza final
@debug: $body
